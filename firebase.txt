todolist.ts

import { Component, inject } from '@angular/core';

import { Firestore, collection, addDoc, collectionData, doc, updateDoc, deleteDoc } from '@angular/fire/firestore';
import { Observable } from 'rxjs';

export class Todolist {
  private firestore: Firestore = inject(Firestore); 
  
  tasks$: Observable<any[]>; 

  constructor() {
    const aCollection = collection(this.firestore, 'tasks');
    this.tasks$ = collectionData(aCollection, { idField: 'id' }); 
  }

  onSubmit(form: NgForm) {
    const aCollection = collection(this.firestore, 'tasks');
    
    addDoc(aCollection, {
      taskName: form.controls['task'].value,
      isCompleted: false
    });

    form.reset();
  }

  onDelete(id: string) {
    const docRef = doc(this.firestore, 'tasks', id);
    deleteDoc(docRef);
  }

  onCheck(id: string, currentStatus: boolean) {
    const docRef = doc(this.firestore, 'tasks', id);
    updateDoc(docRef, { isCompleted: !currentStatus });
  }
}

***Fix error

"Firebase: No Firebase App '[DEFAULT]' has been created - call Firebase App.initializeApp()"

Why This Happens (The Race Condition)
Angular applications rely heavily on Dependency Injection.
When your app boots up, Angular immediately starts creating your Services and Components.
If a Service tries to talk to Firebase the moment it is created, but the initializeApp() function hasn't finished running yet, the app crashes.

Solution: Use the Angular APP_INITIALIZER token to run a function that guarantees the Firebase app is initialized before the main App starts. 

app.config.ts

import { ApplicationConfig, provideBrowserGlobalErrorListeners, APP_INITIALIZER } from '@angular/core';


export const appConfig: ApplicationConfig = {
  providers: [
    provideBrowserGlobalErrorListeners(),
    provideRouter(routes),
    {
      provide: APP_INITIALIZER,
      useValue: () => initializeApp(firebaseConfig),
      multi: true
    },
    provideFirebaseApp(() => getApp()), 
    provideFirestore(() => getFirestore())
  ]
};

todolist.ts
//copy this line and replace the first line of import
import { Component, inject } from '@angular/core';

//copy this part and paste into the beginning import part
import { Firestore, collection, addDoc, collectionData, doc, updateDoc, deleteDoc } from '@angular/fire/firestore';
import { Observable } from 'rxjs';

//copy this and replace the export part
export class Todolist {
  private firestore: Firestore = inject(Firestore); 
  
  tasks$: Observable<any[]>; 

  constructor() {
    const aCollection = collection(this.firestore, 'tasks');
    this.tasks$ = collectionData(aCollection, { idField: 'id' }); 
  }

  onSubmit(form: NgForm) {
    const aCollection = collection(this.firestore, 'tasks');
    
    addDoc(aCollection, {
      taskName: form.controls['task'].value,
      isCompleted: false
    });

    form.reset();
  }

  onDelete(id: string) {
    const docRef = doc(this.firestore, 'tasks', id);
    deleteDoc(docRef);
  }

  onCheck(id: string, currentStatus: boolean) {
    const docRef = doc(this.firestore, 'tasks', id);
    updateDoc(docRef, { isCompleted: !currentStatus });
  }
}

<tbody>
    <tr *ngFor="let t of tasks$ | async">
        <td>{{ t.taskName }}</td>

            <input

                [checked]="t.isCompleted" 
                type="checkbox">
        </td>
        <td>
            <button class="btn btn-danger btn-sm" (click)="onDelete(t.id)">Delete</button>
        </td>
    </tr>
</tbody>
